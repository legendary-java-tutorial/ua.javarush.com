    class Test{
    private int id;
    private Integer id2;

    public Test(int id, Integer id2) {
        System.out.println("Создаем екзепляр Test");
        this.id = id;
        this.id2 = id2;
    }

    public Integer doIt(){
        return id + id2;
    }
}
